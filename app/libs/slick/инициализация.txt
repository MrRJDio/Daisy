    $('.portfolio_slider').slick({
      speed:1000,
      cssEase:'ease-out',
      autoplay: true,
      autoplaySpeed: 3000,
      draggable: false,
      fade:true,
      appendArrows: $('.slide-controls'),
      prevArrow: $('.slide-arrow.arrow-prev'),
      nextArrow: $('.slide-arrow.arrow-next'),
      appendDots: $('.portfolio_slider'), //Где разместить точки
      dots: true,
      dotsClass: 'custom-dots', //slick generates this <ul.custom-dots> within the appendDots container // кастомные точки
    customPaging: function (slider, i) {
        //FYI just have a look at the object to find aviable information
        //press f12 to access the console
        //you could also debug or look in the source
        console.log(slider);
        var slideNumber = (i + 1),
            totalSlides = slider.slideCount;
        return '<a class="dot" role="button" title="' + slideNumber + ' of ' + totalSlides + '"><span class="string">' + slideNumber + '/' + totalSlides + '</span></a>';
    }    
  });
