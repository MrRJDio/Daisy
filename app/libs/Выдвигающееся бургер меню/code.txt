$(document).ready(function(){
	var link = $('.menu-link');  // Обюъявляем переменную
	var link_active = $('.menu-link_active');
	var menu = $('.menu'); 

	link.click(function(){  // Функция клика
		link.toggleClass('menu-link_active'); // При клике добавляем класс на menu-link_active
		menu.toggleClass('menu_active')
	});

	menu.click(function(){
		menu.removeClass('menu_active');  // -Закрывает меню по клику на пункты меню
	});
	menu.click(function(){
		link.removeClass('menu-link_active');  // - Закрывает бургер по клику на пункты меню
	});

$(window).resize(function () {  //-- Закрывает меню при изменении разрешения крана выше 1200
    if($(window).width() >= 1200){
        $('.menu').removeClass('menu_active');
        $('.menu-link').removeClass('menu-link_active');
    };
});

jQuery(function($){
	$(document).mouseup(function (e){ // событие клика по веб-документу или бургеру
		var div = $(".menu_active, .menu-link"); // тут указываем ID элемента исключения
		if (!div.is(e.target) // если цель клика не div имеет в себе e.target(объект функции e)
		    && div.has(e.target).length === 0) {  // или div имеет e.target равное 0, тогда 
			menu.removeClass('menu_active');
			link.removeClass('menu-link_active');
	}
	});
});	

});
